    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE Guild (
    guild_id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    guild_name VARCHAR(30) NOT NULL,
    guild_level INT UNSIGNED DEFAULT 1,
    status TINYINT UNSIGNED DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE `Character` (
    character_id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    adventure_id BIGINT UNSIGNED NOT NULL,
    guild_id BIGINT UNSIGNED NULL,
    job_code INT UNSIGNED NOT NULL,
    nickname VARCHAR(30) NOT NULL,
    state_code INT UNSIGNED DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (guild_id) REFERENCES Guild(guild_id)
);

-- 캐릭터 스탯
CREATE TABLE Character_Stat (
    character_id BIGINT UNSIGNED PRIMARY KEY,

    mp_max INT UNSIGNED NOT NULL,
    mp INT UNSIGNED NOT NULL,

-- 인벤토리
CREATE TABLE Inventory (
